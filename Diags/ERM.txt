Table People {
  id integer [primary key]
  email varchar [unique]
  password varchar
  firstName varchar
  middleName varchar
  lastName varchar
  secondLastName varchar
  major varchar
  role varchar
  age integer
}

Table Posts {
  id integer [primary key]
  message text
  date datetime
  userId integer [not null]
  topicId integer [not null]
}

Table Material {
  id integer [primary key]
  name varchar
  uri varchar
  type varchar
  postId integer [not null]
}

Table Comments {
  id integer [primary key]
  message text
  date datetime
  userId integer [not null]
  postId integer [not null]
}

Table Topics {
  id integer [primary key]
  name varchar
}

Table Followers {
  followerId integer [not null]
  followedId integer [not null]
}

Table Preferences {
  userId integer [not null]
  topicId integer [not null]
}

Table Meetings {
  id integer [primary key]
  title varchar
  description text
  url varchar
  date datetime
  userId integer [not null]
  createdAt datetime
  updatedAt datetime
}

Table Score {
  id_scored integer [not null]
  id_scorer integer [not null]
}
